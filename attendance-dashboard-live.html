  <script>

    const tabs = document.querySelectorAll('.tab');

    const views = document.querySelectorAll('.view');

    const directTabButtons = document.querySelectorAll('[data-tab-target]');

    const statusBox = document.getElementById('status');

    const apiUrlInput = document.getElementById('apiUrl');

    const sheetUrlInput = document.getElementById('sheetUrl');

    const recordsBtn = document.getElementById('recordsBtn');


    function activateTab(name){

      tabs.forEach(t => t.classList.toggle('active', t.dataset.tab === name));

      views.forEach(v => v.classList.toggle('active', v.id === name));

      window.scrollTo({ top: 0, behavior: 'smooth' });

    }

    tabs.forEach(tab => tab.addEventListener('click', () => activateTab(tab.dataset.tab)));

    directTabButtons.forEach(btn => btn.addEventListener('click', () => activateTab(btn.dataset.tabTarget)));


    function pillClass(val){ return val === 'P' ? 'p' : val === 'A' ? 'a' : 'n'; }

    function label(val){ return val === 'P' ? 'Present' : val === 'A' ? 'Absent' : 'No class'; }

    function setStatus(text, mode=''){

      statusBox.textContent = text;

      statusBox.className = 'status' + (mode ? ' ' + mode : '');

    }


    function syncRecordsLink(){

      const url = sheetUrlInput.value.trim();

      recordsBtn.href = url || '#';

      recordsBtn.style.pointerEvents = url && url.startsWith('http') ? 'auto' : 'none';

      recordsBtn.style.opacity = url && url.startsWith('http') ? '1' : '.55';

    }

    sheetUrlInput.addEventListener('input', syncRecordsLink);

    syncRecordsLink();


    function renderDashboard(data){

      const grid = document.getElementById('coursesGrid');

      grid.innerHTML = '';

      Object.entries(data.recentByCourse).forEach(([course, marks]) => {

        const total = data.totals.find(x => x.course === course) || { pct: 0, needed: 0 };

        const card = document.createElement('article');

        card.className = 'course-card';

        card.innerHTML = `

          <div class="course-top">

            <div>

              <div style="font-size:1.05rem;font-weight:800">${course}</div>

              <div style="margin-top:4px;color:var(--muted);font-size:.9rem">Latest logged days</div>

            </div>

            <span class="badge">${total.pct}%</span>

          </div>

          <div class="recent-strip">

            ${marks.map(m => `<div class="pill ${pillClass(m)}" title="${label(m)}">${m}</div>`).join('')}

          </div>

          <div style="margin-top:10px;color:var(--muted);font-size:.9rem">

            Needed for 75%: <strong style="color:var(--text)">${total.needed}</strong>

          </div>

        `;

        grid.appendChild(card);

      });

    }


    function renderTotals(data){

      const body = document.getElementById('totalsBody');

      body.innerHTML = data.totals.map(row => {

        const pctClass = row.pct >= 75 ? 'good' : row.pct >= 60 ? 'mid' : 'bad';

        return `

          <tr>

            <td style="font-weight:800">${row.course}</td>

            <td>${row.conducted}</td>

            <td>${row.present}</td>

            <td>${row.absent}</td>

            <td class="pct ${pctClass}">${row.pct}%</td>

            <td style="font-weight:800">${row.needed}</td>

          </tr>

        `;

      }).join('');

    }


    function renderStats(data){

      document.getElementById('courseCount').textContent = data.totals.length || '0';

      document.getElementById('recentDate').textContent = data.latestDate || '—';

      const avg = data.totals.length

        ? data.totals.reduce((a,b) => a + Number(b.pct || 0), 0) / data.totals.length

        : 0;

      document.getElementById('avgAttendance').textContent = `${Math.round(avg)}%`;

    }


    async function loadLiveData(){

      const url = apiUrlInput.value.trim();

      syncRecordsLink();

      if (!url.startsWith('http')) {

        setStatus('Paste a valid Apps Script web app URL first.', 'err');

        return;

      }

      setStatus('Loading live data...');

      try {

        const res = await fetch(url + (url.includes('?') ? '&' : '?') + 'mode=dashboard');

        if (!res.ok) throw new Error('HTTP ' + res.status);

        const data = await res.json();

        if (!data || !data.totals || !data.recentByCourse) throw new Error('Unexpected JSON shape');

        renderDashboard(data);

        renderTotals(data);

        renderStats(data);

        setStatus('Live data loaded successfully.', 'ok');

      } catch (err) {

        setStatus('Could not load live data. Check deployment access, URL, and JSON code. ' + err.message, 'err');

      }

    }


    document.getElementById('loadBtn').addEventListener('click', loadLiveData);

  </script>
